Closest_Pair_of_Points function(space class)
	if len(list_of_points) >= 2
		point1 
		point2 
		closest_distance = sqrt( (list_of_points[0][0] - list_of_points[1][0])^2 - (list_of_points[0][1] - list_of_points[1][1])^2 ) 
		for i in space.list_of_points
			for j in space.list_of_points
			dist_points = sqrt( (list_of_points[i][0] - list_of_points[j][0])^2 - (list_of_points[i][1] - list_of_points[j][1])^2 ) 
			if closest_distance >= dist_points  
				closest_distance = dist_points
				set point1 as list_of_points[i]
				set point2 as list_of_points[j] 
		print(point1, point2, closest_distance) 
	else print "needs more than two points"
